Drawing Dragon Curve using L-system

Reference: http://en.wikipedia.org/wiki/Dragon_curve

To compile the program below : 

clang++ -o dragon_curve dragon_curve.cpp -lglfw -framework Cocoa -framework OpenGL -framework IOKit -framework CoreVideo

 20 //angle: 120°
 21 // initial string F
 22 // string rewriting rules
 23 // F ↦ F+F−F
 24 // - turn left by 90 degrees
 25 // + turn right by 90 degrees
 26 std::string terDragonLSystemRepresentation() {
 27   int generation = 0;
 28   std::string current = "F";
 29 
 30   while(generation < 14) {
 31     std::string next;
 32     for(std::string::iterator it = current.begin(); it != current.end(); it++) {
 33       switch(*it) {
 34         case 'F':
 35         next.append("F+F-F");
 36         break;
 37         case '-':
 38         next.append("-");
 39         break;
 40         case '+':
 41         next.append("+");
 42         break;
 43       };
 44     }
 45 
 46     current = next;
 47     generation++;
 48     //std::cout << "Generation: " <<  " " << current << std::endl;
 49   }
 50 
 51   return current;
 52 }

 62 std::string twinDragonLSystemRepresentation() {

 97 // Heighway dragon
 98 // angle: 90°
 99 // initial: string FX
100 // string rewriting rules
101 // X ↦ X+YF+
102 // Y ↦ −FX−Y.
103 // - turn left by 90 degrees
104 // + turn right by 90 degrees
105 std::string heighwayDragonLSystemRepresentation() {
106   int generation = 0;
107   std::string current = "FX";
108 
109   while(generation < 21) {
110     std::string next;
111     for(std::string::iterator it = current.begin(); it != current.end(); it++) {
112       switch(*it) {
113         case 'X':
114         next.append("X+YF+");
115         break;
116         case 'Y':
117         next.append("-FX-Y");
118         break;

121         break;

124         break;

132     generation++;

139 void drawDragon(int type, float x, float y) {
140   std::string lrep;
141   float angle;
142 
143   switch(type) {
144     case 0:
145     lrep = heighwayDragonLSystemRepresentation();
146     angle = 90.0f;
147     break;
148     case 1:
149     lrep = twinDragonLSystemRepresentation();
150     angle = 90.0f;
151     break;
152     case 2:
153     lrep = terDragonLSystemRepresentation();
154     angle = 120.0f;
155     break;
156   };
157 
158 
159   glClear(GL_COLOR_BUFFER_BIT);
160   glPushMatrix();
161   for(std::string::iterator it = lrep.begin(); it != lrep.end(); it++) {
162     if(*it == 'F') {
163       glColor4f(0.5f, 1.0f, 0.0f, 1.0f);
164       glLineWidth(2.0f);
165       glBegin(GL_LINES);
166       glVertex2f(x, y);
167       x = x + LINE_LENGTH;
168       glVertex2f(x, y);
169       glEnd();
170     }
171     if(*it == '-' || *it == '+') {
172       glTranslatef(x, y, 0.0f);
173       x = 0.0f;
174       y = 0.0f;
175       int sign = ((*it == '-') ? 1 : -1);
176       glRotatef(sign * angle, 0.0f, 0.0f, 1.0f);
177     }
178   }
179   glPopMatrix();
180 }

182 int main(int argc, char *argv[]) {
183   GLFWwindow* window;
184 
185   if(!glfwInit()) {
186     return -1;
187   }
188 
189   GLFWmonitor* monitor  = glfwGetPrimaryMonitor();
190   const GLFWvidmode* vidmode = glfwGetVideoMode(monitor);
191 
192   window = glfwCreateWindow(vidmode->width, vidmode->height, "Sierpinski Triangle", monitor, NULL);
193   if(!window) {
194     glfwTerminate();
195     return -1;
196   }
197 
198   glfwSetKeyCallback(window, key_callback);
199   glfwMakeContextCurrent(window);
200 
201   int width, height;
202 
203   glfwGetFramebufferSize(window, &width, &height);
204   glViewport(0, 0, width, height);
205   glMatrixMode(GL_PROJECTION);
206   glLoadIdentity();
207   glOrtho(0.0f, width, 0.0f, height, 0.0, 1.0f);
208   glMatrixMode(GL_MODELVIEW);
209   glLoadIdentity();
210 
211   long int count = 0;
212   while(!glfwWindowShouldClose(window)) {
213     if(count > 100) {
214       count = 0;
215     }
216 
217     switch(++count % 3) {
218       case 0:
219       drawDragon(0, width / 2, height / 2);
220       break;
221       case 1:
222       drawDragon(1, width / 2, height / 2);
223       break;
224       case 2:
225       drawDragon(2, width / 2, height - 200.0f);
226       break;
227     };
228     glfwSwapBuffers(window);
229     glfwPollEvents();
230   }
231 
232   glfwTerminate();
233   return 0;
234 }
