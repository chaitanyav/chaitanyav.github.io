Pattern formed by drawing circles recursively

Reference: The Nature of Code: Simulating Natural Systems with Processing Paperback â€“ December 13, 2012 by Daniel Shiffman

To compile the program below : 

clang -o circle_recursive circle_recursive.c -lglfw -framework Cocoa -framework OpenGL -framework IOKit -framework CoreVideo

 1 /* Author: NagaChaitanya Vellanki */
 2 
 3 #include <GLFW/glfw3.h>
 4 #include <stdio.h>
 5 #include <stdlib.h>
 6 #include <math.h>
 7 
 8 static float PI = 3.14159265358979323846;
 9 static float RADIUS = 300.0f;
10 
11 void draw_circle(float x, float y, float radius) {
12   glColor4f(0.5f, 1.0f, 0.0f, 1.0f);
13   glPointSize(2.0f);
14   glBegin(GL_POINTS);
15   for (int radian = 0; radian < 360; radian++)
16     {
17       float degree = radian * (PI / 180.0);
18       glVertex2f(x + radius * cosf(degree), y + radius * sinf(degree));
19     }
20 
21     glEnd();
22     if(radius > 10.0f) {
23       draw_circle(x + radius, y, radius / 2);
24       draw_circle(x - radius, y, radius / 2);
25       draw_circle(x, y + radius, radius / 2);
26       draw_circle(x, y - radius, radius / 2);
27     }
28   }
29 
30   static void key_callback(GLFWwindow* window, int key, int scancode, int action, int mods)
31   {
32     if (key == GLFW_KEY_ESCAPE && action == GLFW_PRESS)
33       glfwSetWindowShouldClose(window, GL_TRUE);
34     }
35 
36     int main(int argc, char *argv[]) {
37       GLFWwindow* window;
38 
39       if(!glfwInit()) {
40         return -1;
41       }
42 
43       GLFWmonitor* monitor  = glfwGetPrimaryMonitor();
44       const GLFWvidmode* vidmode = glfwGetVideoMode(monitor);
45 
46       window = glfwCreateWindow(vidmode->width, vidmode->height, "Circle Recursive", monitor, NULL);
47       if(!window) {
48         glfwTerminate();
49         return -1;
50       }
51 
52       glfwSetKeyCallback(window, key_callback);
53       glfwMakeContextCurrent(window);
54 
55       int width, height;
56       glfwGetFramebufferSize(window, &width, &height);
57       float x = width / 2;
58       float y = height / 2;
59 
60       glViewport(0, 0, width, height);
61       glMatrixMode(GL_PROJECTION);
62       glLoadIdentity();
63       glOrtho(0.0f, width, 0.0f, height, 0.0, 1.0f);
64       glMatrixMode(GL_MODELVIEW);
65       glLoadIdentity();
66 
67       while(!glfwWindowShouldClose(window)) {
68         glClear(GL_COLOR_BUFFER_BIT);
69         draw_circle(x, y, RADIUS);
70         glfwSwapBuffers(window);
71         glfwPollEvents();
72       }
73 
74       glfwTerminate();
75       return 0;
76     }
