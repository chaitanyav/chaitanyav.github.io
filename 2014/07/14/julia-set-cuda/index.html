Julia set using CUDA, GLFW

Reference: Chapter-04 of CUDA by Example: An Introduction to General-Purpose GPU Programming 1st Author: Jason Sanders, Edward Kandrot Pages: 312 Publisher: Addison-Wesley Professional Â©2010 ISBN: 0131387685 9780131387683

To compile the program below : 

nvcc julia_set.cu -lglfw -Xlinker -framework,Cocoa,-framework,OpenGL,-framework,IOKit,-framework,CoreVideo

  9 struct cuComplex {
 10   float r;
 11     float i;
 12 
 13   __device__ cuComplex(float a, float b) : r(a), i(b) {
 14   }
 15 
 16   __device__ float magnitude2(void) {
 17     return r * r + i * i;
 18   }
 19 
 20   __device__ cuComplex operator*(const cuComplex& a) {
 21     return cuComplex(r * a.r - i * a.i, i * a.r  + r * a.i);
 22   }
 23 
 24   __device__ cuComplex operator+(const cuComplex& a) {
 25     return cuComplex(r + a.r, i + a.i);
 26   }
 27 };
 28 
 29 __device__ int julia(int x, int y) {
 30   const float scale = 1.5;
 31   int dim_2 = DIM / 2;
 32   float jx = scale * (float) (dim_2  - x) / dim_2;
 33   float jy = scale * (float) (dim_2  - y) / dim_2;
 34 
 35   cuComplex c(-0.8, 0.156);
 36   cuComplex a(jx, jy);
 37 
 38   for(int i = 0; i < 200; i++) {
 39     a = a * a + c;
 40     if (a.magnitude2() > 1000) {
 41       return 0;
 42       }
 43   }
 44   return 1;
 45 }

 47 __global__ void kernel(unsigned char *ptr) {
 48   int x = blockIdx.x;
 49   int y = blockIdx.y;
 50   int offset = x + y * gridDim.x;
 51   int juliavalue = julia(x, y);
 52   ptr[offset * 4 + 0] = 127 * juliavalue;
 53   ptr[offset * 4 + 1] = 255 * juliavalue;
 54   ptr[offset * 4 + 2] = 0;
 55   ptr[offset * 4 + 3] = 255;
 56 }

 58 struct CPUBitmap {
 59   unsigned char    *pixels;
 60   int     x, y;
 61   void    *dataBlock;
 62 
 63   CPUBitmap( int width, int height, void *d = NULL  ) {
 64     pixels = new unsigned char[width * height * 4];
 65     x = width;
 66     y = height;
 67     dataBlock = d;
 68   }
 69 
 70   ~CPUBitmap() {
 71     delete [] pixels;
 72   }
 73 
 74   unsigned char* get_ptr( void  ) const   { return pixels;  }
 75   long image_size( void  ) const { return x * y * 4;  }
 76 
 77   void display_and_exit() {
 78     CPUBitmap**   bitmap = get_bitmap_ptr();
 79     *bitmap = this;
 80     GLFWwindow* window;
 81 
 82     if(!glfwInit()) {
 83       printf("Failed on GLFW init\n");
 84       exit(EXIT_FAILURE);
 85     }
 86 
 87     window = glfwCreateWindow(x, y, "Julia Set", NULL, NULL);
 88     if(!window) {
 89       glfwTerminate();
 90       printf("Failed to create window\n");
 91       exit(EXIT_FAILURE);
 92     }
 93 
 94     glfwMakeContextCurrent(window);
 95     glfwSetKeyCallback(window, key_callback);
 96 
 97     while(!glfwWindowShouldClose(window)) {
 98       CPUBitmap*   bitmap =  *(get_bitmap_ptr());
 99       glClearColor( 0.0, 0.0, 0.0, 1.0  );
100       glClear( GL_COLOR_BUFFER_BIT );
101       glDrawPixels(bitmap->x, bitmap->y, GL_RGBA, GL_UNSIGNED_BYTE, bitmap->pixels);
102       glfwSwapBuffers(window);
103       glfwPollEvents();
104     }
105     glfwTerminate();
106   }
107 
108     // static method used for glut callbacks
109   static CPUBitmap** get_bitmap_ptr( void  ) {
110     static CPUBitmap   *gBitmap;
111     return &gBitmap;
112   }
113 
114   static void key_callback(GLFWwindow* window, int key, int scancode, int action, int mods)
115   {
116     if (key == GLFW_KEY_ESCAPE && action == GLFW_PRESS){
117       glfwSetWindowShouldClose(window, GL_TRUE);
118       glfwTerminate();
119       exit(EXIT_SUCCESS);
120     }
121   }
122 };

124 void handleError(cudaError_t error, int lineNo) {
125   if(error != cudaSuccess) {
126     printf("Error: %s\n in file %s at line no %d\n",
127     cudaGetErrorString(error), __FILE__, __LINE__);
128     exit(EXIT_FAILURE);
129   }
130 }

132 int main(int argc, char *argv[]) {
133   CPUBitmap bitmap(DIM, DIM);
134   unsigned char *dev_bitmap;
135   handleError(cudaMalloc((void **) &dev_bitmap, bitmap.image_size()), __LINE__);
136   dim3 grid(DIM, DIM);
137   kernel<<<grid,1>>>(dev_bitmap);
138   handleError(cudaMemcpy(bitmap.get_ptr(), dev_bitmap,bitmap.image_size(), cudaMemcpyDeviceToHost), __LINE__);
139   bitmap.display_and_exit();
140   handleError(cudaFree(dev_bitmap), __LINE__);
141   return 0;
142 }
